function crc32(b:byte; crc:longint):longint;
begin
  crc32:=crc32tab[byte(crc xor longint(b))] xor ((crc shr 8) and $00ffffff);
end;

function crcend(crc:longint):longint;
begin
  crcend:=crc xor $ffffffff;
end;

function showcrc(i:longint):string; assembler;
asm
  jmp    @1
  @0:
  db     '0123456789abcdef'
  @1:
  cld
  les    di, @result
  mov    ax, 0008h
  stosb
  mov    cx, 4
  mov    si, 3
  @2:
  mov    al, byte [I+si]
  dec    si
  push   si
  mov    bl, al
  mov    dl, bl
  and    bx, 00f0h
  shr    bx, 4
  and    dx, 000fh
  mov    si, bx
  mov    al, byte [cs:@0+si]
  stosb
  mov    si, dx
  mov    al, byte [cs:@0+si]
  stosb
  pop    si
  loop   @2
end;

var crc:longint; f:file; buf:array[1..1024] of byte; nr,n:word;

begin
  assign(f,'TEST.EXE');              { calculate the CRC of "TEST.EXE" }
  {$I-} reset(f,1); {$I+}            { open the file }
  if ioresult=0 then begin           { if it exists, then start calculating }
    crc:=$ffffffff;                  { first, initialize the CRC variable }
    repeat
      blockread(f,buf,sizeof(buf),nr);
      for n:1 to nr do crc:=crc32(buf[n],crc); { add each byte into CRC var }
    until nr=0;
    close(f);                        { close the file }
    crc:=crcend(crc);                { make few last adjustments to CRC var }
    writeln('CRC is '+showcrc(crc)); { write CRC to the screen }
  end else writeln('File not found.');
end.
