FUNCTION ROM_Bios_Date : String8;
{ DESCRIPTION:
    Gets the ROM BIOS date in a ASCII string.
  SAMPLE CALL:
    S := ROM_Bios_Date
  RETURNS:
    e.g., '06/10/85'.
  NOTES:
    Later versions of the Compaq have this release date shifted by 1 byte,
      to start at F000:FFF6h }

var
  Tmp : String8;

BEGIN { ROM_Bios_Date }
  FillChar(Tmp,0,8);
  Tmp[0] := Chr(8);
  Move(Mem[$F000:$FFF5],Tmp[1],8);
  ROM_Bios_Date := Tmp;
END; { ROM_Bios_Date }
