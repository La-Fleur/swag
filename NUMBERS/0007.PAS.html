Type
  bsingle = Array [0..3] of Byte;

{ converts Microsoft 4 Bytes single to TP Real }

Function msb_to_Real (b : bsingle) : Real;
Var
  pReal : Real;
  r     : Array [0..5] of Byte Absolute pReal;
begin
  r [0] := b [3];
  r [1] := 0;
  r [2] := 0;
  move (b [0], r [3], 3);
  msb_to_Real := pReal;
end; { Function msb_to_Real }

}
