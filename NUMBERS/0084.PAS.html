Function SwapBin (Chiffre : Byte) : Byte;

{ Usefull when you must mirror a matrix }

Var I, Temp : Byte;
    Tempo   : Byte;

Begin

   Tempo := 0;

   For I := 7 Downto 0 do Begin
       Temp  := (Chiffre and (1 shl Abs(I-7)));
       If Temp = 0 then Tempo := (Tempo shl 1)
       Else Tempo := (Tempo shl 1) + 1;
   End;

   SwapBin := Tempo;

End;
