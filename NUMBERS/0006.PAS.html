Function rolW(b : Word; n : Byte) : Word; Assembler;
Asm
  mov ax, b
  mov cl, n
  rol ax, cl
end;

Function rolB(b, n : Byte) : Byte; Assembler;
Asm
  mov al, b
  mov cl, n
  rol al, cl
end;

Function rolW1(b : Word) : Word; Assembler;
Asm
  mov ax, b
  rol ax, 1
end;

  $D1/$C0);     {rol ax,1}
