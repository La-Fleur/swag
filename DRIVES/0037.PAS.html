FUNCTION Verify_State : boolean;
{ DESCRIPTION:
    Gets disk verify state flag.
  SAMPLE CALL:
    B := Verify_State;
  RETURNS:
    TRUE  = on: verify after write
    FALSE = off: no verify after write }

var
  HTregs : registers;

BEGIN { Verify_State }
  HTregs.AH := $54;
  MsDos(HTregs);
  Verify_State := HTregs.AL = $01;
END; { Verify_State }
