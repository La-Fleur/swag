FUNCTION First_CD_ROM_Drive : byte;
{ DESCRIPTION:
    Gets the first installed CD-ROM drive letter in a system.
  SAMPLE CALL:
    NB := First_CD_ROM_Drive;
  RETURNS:
    0 : drive A
    1 : drive B
    and so on... }

var
  HTregs : registers;

BEGIN { First_CD_ROM_Drive }
  HTregs.AX := $1500;
  HTregs.BX := $0000;
  Intr($2F,HTregs);
  First_CD_ROM_Drive := HTregs.CL;
END; { First_CD_ROM_Drive }
