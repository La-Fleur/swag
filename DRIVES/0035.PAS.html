FUNCTION Startup_Drive : byte;
{ DESCRIPTION:
    Gets the startup (boot) drive.
  SAMPLE CALL:
    NB := Startup_Drive;
  RETURNS:
    1 : drive A
    2 : drive B
    and so on... }

var
  HTregs : registers;

BEGIN { Startup_Drive }
  HTregs.AX := $3305;
  MsDos(HTregs);
  Startup_Drive := HTregs.DL;
END; { Startup_Drive }
