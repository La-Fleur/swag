Uses
  Dos;

Function driveexist(ch : Char) : Boolean;
begin
  DriveExist := disksize(ord(upcase(ch)) - 64) <> - 1;
end;


{ Kerry Sokalsky }

Const
  exist : Boolean  = True;
  ch    : Integer  = 67;   { 'C' - Skip floppy Drives (A&B) }
  lastdrive : Char = ' ';

begin
  While LastDrive = ' ' do
  begin
    if driveexist(Chr(ch)) then
      Inc(Ch)
    else
      LastDrive := Chr(Ch - 1);
  end;

  Writeln(LastDrive);
end.
