function DiskChange( DriveNmber :Byte) :Boolean;
Begin
 ASm
   Mov AH, $16
   Mov DL, driveNmber
   Int $13
   Mov AL,AH;  { use AL & AH as a Return Value }
 End;
End;

Begin
  If DiskChange(0) then Write(' Disk has Changed in Drive ''A'' ')
   Else
     Write(' Disk Has changed ');
end.
