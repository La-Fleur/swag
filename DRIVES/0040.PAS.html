type
  TInfoBuffer = record
    InfoLevel : word; {may be 0}
    Serial    : longInt;
    VolLabel  : array [0..10] of char;
    FileSystem: array [0..7] of char;
  end;

function GetSerial(DiskNum : Byte; var I : TInfoBuffer) : word; assembler;
asm
  mov  ah, 69h
  mov  al, 00h
  mov  bl, DiskNum
  push ds
  lds  dx, I
  int  21h
  pop  ds
  jc   @bad
  Xor  ax, ax
 @bad:
end;

function SetSerial(DiskNum : Byte; var I : TInfoBuffer) : word; assembler;
asm
  mov  ah, 69h
  mov  al, 01h
  mov  bl, DiskNum
  push ds
  lds  dx, I
  int  21h
  pop  ds
  jc   @bad
  xor  ax, ax
 @bad:
end;
