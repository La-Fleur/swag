FUNCTION CD_ROM_Units : byte;

{ DESCRIPTION:
    Gets the number of installed CD-ROM drives in a system.
  SAMPLE CALL:
    NB := CD_ROM_Units;
  RETURNS:
    0    : driver not installed
    else : number of CD-ROM units }

var
  HTregs : registers;

BEGIN { CD_ROM_Units }
  HTregs.AX := $1500;
  HTregs.BX := $0000;
  Intr($2F,HTregs);
  CD_ROM_Units := HTregs.BL;
END; { CD_ROM_Units }
