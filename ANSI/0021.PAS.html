Uses
  Dos, Crt;

Var
  Old_Vect : Pointer;

Procedure Redirect_OutPut(Character : Char); Interrupt;
begin
  Write(Character);
end;

begin
 GetIntVec($29, Old_Vect);        { Save Old Vector }
 SetIntVec($29, @Redirect_OutPut);

 Exec(' MainProgram ','');        { all output using WriteLn(Con,'????') }
