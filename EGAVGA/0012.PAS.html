{
Here's a quick proc. to return the current video mode:
}

Uses
  Dos;

Function CurVidMode : Byte;

Var
  Regs : Registers;

begin;

  Regs.Ah :=$f;
  Intr($10, Regs);
  CurVidMode := Regs.Al;

end;

begin
  Writeln(CurVidMode);
end.
