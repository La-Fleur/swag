function Edit(No : byte) : Char;
var x, y : byte;
    car  : Char;
    Sortie : Boolean;
    Go     : byte;
begin
   UpDate(No);
   x := 0;
   y := 0;
   Sortie := false;

   repeat
      GotoXY(32 + 2*x,5+y);
      HighVideo;
      Write(Buffer[x,y]);
      GotoXY(32 + 2*x,5+y);
      repeat
      until keypressed;
      car := ReadKey;
      GotoXY(32 + 2*x,5+y);
      LowVideo;
      Write(Buffer[x,y]);
      if (car = 'q') or (car = 'Q') then car := #13;
      if car = #0 then car := ReadKey;
      case car of
         '2',chr(80)   : if y = 8 then y := 0 else inc(y);
         '8',chr(72)   : if y = 0 then y := 8 else dec(y);
         '4',chr(75)   : if x = 0 then x := 7 else dec(x);
         '6',chr(77)   : if x = 7 then x := 0 else inc(x);
         ' '           : if Buffer[x,y] = '*' then Buffer[x,y] := 'ú' else
Buffer[x,y] := '*';
         #13, #81, #73 : Sortie := true;
         #27           : Sortie := True;
         'G','g'       : begin
                            GotoXY(20,24);
                            Write('Aller Ö quel code ascii ? ');
                            Read(Go);
                            if (Go >= 32) and (go <= 180) then
                            begin
                               Encode(No);
                               EnCours := Go -1;
                               Car := #81;
                               Sortie := true;
                            end;
                            GotoXY(1,24);
                            ClrEol;
                         end;

         'v'           : begin
                           Preview;
                           update(No);
                         end;
         'a'           : begin
                            Encode(No);
                            PreviewAll;
                            Bufferize(No);
                            Update(No);
                         end;
      end;
   until (sortie);
   Encode(No);
   Edit := Car;
end;

      GotoXY(4 + 2*x,5+y);

      GotoXY(4 + 2*x,5+y);
