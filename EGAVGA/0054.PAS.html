type
  Font_Type  = array[1..4096] of byte;

PROCEDURE Font_Get(var Fnt : Font_Type);

{ DESCRIPTION:
    Get the active font table in buffer #0.
  SAMPLE CALL:
    Font_Get(Font_Table);
  RETURNS:
    The font table.
  NOTES:
    Works in VGA only, and with 8x16 fonts }

var
  Regs : registers;

BEGIN { Font_Get }
  Regs.AH := $11;
  Regs.AL := $30;
  Regs.BH := 6;                        { VGA: 8 x 16 }
  Intr($10,Regs);
  Move(Mem[Regs.ES:Regs.BP],Fnt,4096);
END; { Font_Get }
