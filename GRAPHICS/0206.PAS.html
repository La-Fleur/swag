 Procedure Putpixel(x,y : integer; COL : byte);
 BEGIN
    ASM
        Push    es
        mov     ax, $a000
        mov     es, ax
        Xor     di, di
        mov     ax, [x]
        mov     bx, 320
        mul     bx
        add     ax, [y]
        mov     di, ax
        mov     ah, [col]
        mov     es:[di], ah
        pop     es
    end
end;
