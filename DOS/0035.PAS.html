program Reboot;
begin
  Inline  ($EA/$F0/$FF/$00/$F0)
end.

Procedure ColdBoot;  Assembler;
   Asm
      Xor  AX, AX
      Mov  ES, AX
      Mov  Word PTR ES:[472h],0000h   {This is NOT a WARM boot}
      Mov  AX, 0F000h
      Push AX
      Mov  AX, 0FFF0h
      Push AX
      Retf
   End;
