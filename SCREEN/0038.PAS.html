Procedure ClrScr(attr : Byte; ch : Char); Assembler;
Asm
  mov  ax, $b800
  mov  es, ax
  xor  di, di
  mov  cx, 80*25
  mov  ah, attr
  mov  al, &ch
  rep  stosw
end;
