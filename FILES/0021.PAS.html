PROCEDURE Get_File_Size(FName : string;
                    var FSize : longint;
                    var Error : word);
{ DESCRIPTION:
    Gets size of existing file, in bytes.
  SAMPLE CALL:
    Get_File_Size('C:\COMMAND.COM',FSize,Error);
  RETURNS:
    FSize : 0 if error
            else file size
    Error : DosError code }

var
  SR    : SearchRec;

BEGIN { Get_File_Size }
  {$I-}
  FindFirst(FName,Archive,SR);
  Error := DosError;
  {$I+}
  if Error = 0 then
    FSize := SR.Size
  else
    FSize := 0;
END; { Get_File_Size }
