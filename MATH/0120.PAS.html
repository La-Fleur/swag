Procedure Drawon(xt,yt:real);
Begin
  PutPixel(trunc(320-(10*xt)),trunc(240-(10*yt)),Yellow);
end;


Procedure grafico;
var x0,y0,x,y : real;
begin
  x := -100; { Calculates x from -100 to 100 (*) }
  drawon(c,0);
  repeat
    x := x + 0.01; { Precision }
    y := (a*(x*x)) + (b*x) + (c);
    drawon(x,y);
  until x >= 100 { Calculates "x" from -100 to 100 (*) }
end;

Procedure DrawCartesian;

Begin
