FUNCTION LogTen( X: REAL ) : REAL;
BEGIN (* LogTen *)

   IF X <= 0.0 THEN
      LogTen := 0.0
   ELSE
      LogTen := LN( X ) * (1/ln(10));
end;

FUNCTION Power( x: REAL; y: REAL ) : REAL;
BEGIN (* Power *)
   IF x > 0 THEN
       Power := EXP( y * LN( x ) )
   ELSE
      Power := 0.0;
END   (* Power *);
