function gcd(a, b : word) : word; assembler;
asm
  mov     ax,a
  mov     bx,b
@start:
  or      bx,bx
  jz      @endgcd
  xor     dx,dx
  div     bx
  mov     ax,bx
  mov     bx,dx
  jmp     @start
@endgcd:
end;
