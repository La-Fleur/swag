Procedure Power(Var Num,Togo,Sofar:LongInt);

Begin
  If Togo = 0 then
    Exit;
  If Sofar = 0 then
    Sofar := num
  Else
    Sofar := Sofar*Num;
  Togo := Togo-1;
  Power(Num,Togo,Sofar)
End;
