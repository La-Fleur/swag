Function  Is4dos : Boolean;Assembler;
 asm
    MOV AX, 0D44DH
    XOR BX,BX
    INT 2FH
    CMP AX,44DDH
    JE  @OK
    MOV AL,0
    JMP @END
    @OK: MOV AL,1
    @END:
 end;

asm
 mov ah,0Fh
 int 10h
 xor ah,ah
 int 10h;
end;

begin
asm
 xor ah,ah

 mov asciik,al
end;
