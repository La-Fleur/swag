uses oKor, oStack_b;
type MyKor=object(Kor)
           Procedure Triangle(n:integer; a:real);
           End;

procedure MyKor.Triangle(n:integer; a:real);
var v:integer;
    s:Stack;
begin
  with s do
    begin
      init;
      push(1,n,a);
      while not empty do
        begin
          pop(v,n,a);
          case v of
            1:     if n=0 then
                   else
                     begin
                       Dopredu(a); Vlavo(120);
                       push(2,n,a);
                       push(1,n-1,2*a/5);
                     end;
            2,3: begin
                     Dopredu(a); Vlavo(120);
                     if v<3 then push(v+1,n,a);
                     push(1,n-1,2*a/5);
                   end;
          end {case}
        end {while}
    end {with}
end;

var k:MyKor;

begin

  with k do Begin
      Init(200,-230,-30);
      Triangle(6,500);
      PresunXY(-300,230); Pis('Squards level 6 - Norekusion');
      CakajKlaves;
      Koniec;
            End

End.
