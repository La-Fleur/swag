Uses Okor;

Type Mykor=Object(kor)
             Procedure Triangle(n:integer;s:real);
             End;

  procedure Mykor.Triangle(n:integer;s:real);
  var
    i:integer;
  begin
    if n=0 then
    else for i:=1 to 3 do
    begin
      Zmenfp(i);
      dopredu(s);
      vlavo(120);
      Triangle(n-1,s/2.5);
    end;
  end;

  Var k:Mykor;
  Begin

  With k do Begin
            Init(240,-240,-30);
            Triangle(6,480);
            CakajKlaves;
            Koniec;
            End;

  End.
