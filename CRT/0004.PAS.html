FillChar(Mem[$A000:$0000],320*200,0);

{ You can double speed by using 16 bit wide data transfer: }

Procedure FillChar16(Var X;Count : Word;Value : Byte); Assembler;
Asm
  les   di,X
  mov   cd,Count
  shr   cx,1
  mov   al,Value
  mov   ah,al
  rep   stosw
  test  Count,1
  jz    @end
  stosb
@end:
end;
