UNIT CCHAR;

uses crt;

CONST vidseg:word=$B800;

var c:char;

procedure checkvidseg;
begin
  if (mem[$0000:$0449]=7) then vidseg:=$B000 else vidseg:=$B800;
end;

function currentchar:char;
begin
      checkvidseg;
      inline($FA);
      currentchar:=chr(mem[vidseg:(160*(wherey-1)+2*(wherex-1))]);
      inline($FB);
end;

end.
