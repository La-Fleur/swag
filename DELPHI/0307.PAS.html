You want GetDeviceCaps() instead, try this:

var
  VertPix, HorzPix : integer ;
begin
  VertPix := GetDeviceCaps( Printer.Canvas.Handle,LOGPIXELSX ) ;
  HorzPix := GetDeviceCaps( Printer.Canvas.Handle,LOGPIXELSY ) ;
