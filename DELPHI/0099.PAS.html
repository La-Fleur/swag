procedure TEditForm.SetEditRect;
Var
  R : TRect;
begin
  R := DisplayMemo.ClientRect;
  R.Left:=R.Left + kMemoIndent;
  R.Top:=R.Top + 2;
  R.Bottom:=R.Bottom - 2;
  R.Right:=R.Right - kMemoIndent;
  SendMessage(DisplayMemo.Handle, EM_SETRECT, 0, Longint(@R));
end;
