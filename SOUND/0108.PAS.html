Program Bomb;
Uses crt;
Var w : word;
     c : char;
Procedure space;
Begin
 w := 10;
 Repeat
   Inc(w,5);
   Sound(w*w);
   Delay(10);
   Sound(w*10);
   Delay(5);
 Until keypressed;
 NoSound;
End;
Procedure deep;
Var e : longint;
Begin
 e := 10;
 Repeat
  iNC(e,1000);
  Sound(Round(  (1/e*100)*100000));
  Delay(1);
 Until keypressed;
 NoSound;
End;
